# Task: Write desktop modal rendering tests

## Description
Write behavioral tests for TaskFormSheet to verify the modal renders correctly on desktop with proper flexbox centering and width constraints.

## Acceptance Criteria
- [x] Test: modal has flexbox centering container
- [x] Test: modal content has correct width constraints (w-[min(95vw,600px)])
- [x] Tests verify CSS classes are applied correctly

## Technical Details
- File: `frontend/src/components/dialogs/tasks/__tests__/TaskFormSheet.test.tsx`

```typescript
describe('TaskFormSheet Behavioral Tests', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  describe('Desktop Modal Rendering', () => {
    it('renders modal with flexbox centering container', async () => {
      const { TaskFormSheetImpl } = await import('../TaskFormSheet');
      const { container } = render(<TaskFormSheetImpl mode="create" projectId="test-123" />);
      const flexContainer = container.querySelector('.fixed.inset-0.flex.items-start.justify-center');
      expect(flexContainer).toBeInTheDocument();
    });

    it('renders modal content with correct width constraints', async () => {
      const { TaskFormSheetImpl } = await import('../TaskFormSheet');
      render(<TaskFormSheetImpl mode="create" projectId="test-123" />);
      const modal = screen.getByTestId('task-form-sheet');
      expect(modal).toHaveClass('w-[min(95vw,600px)]');
    });
  });
});
```
